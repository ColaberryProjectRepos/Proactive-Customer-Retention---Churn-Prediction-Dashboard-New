Creating measures

1.From Power BI desktop ribbon, Home tab click 'New Measure'
2. Fill in all the DAX formulas provided below, one at a time, and when you finish Press Enter or select Commit (checkmark icon) in the formula bar.
3. After creating each measure choose the necessary format and decimal for each value (measure) then save.




Follow the steps in the GIF below:




Measures DAX Formulas
1. Active Cust =
CALCULATE(
COUNTROWS('electronics_data'),
FILTER(
'electronics_data',
'electronics_data'[.ChurnStatusByDuration1] = "Active" &&
YEAR('electronics_data'[Order Date]) = 2023 &&
NOT ISBLANK('electronics_data'[City])
)
)
2. Churn Rate =
DIVIDE(
[.Churned Cust],
[.Churned Cust] + [.Active Cust],
0
)
3.Churned Cust =
CALCULATE(
COUNTROWS('electronics_data'),
FILTER(
'electronics_data',
'electronics_data'[.ChurnStatusByDuration1] = "Churned" && -- Filter where ChurnStatusByDuration is Churned
YEAR('electronics_data'[Order Date]) = 2023 && -- Ensure we are only considering the year 2023
NOT ISBLANK('electronics_data'[City]) -- Exclude rows with blank cities
)
)
4.ChurnProbability2023 =
SWITCH(
TRUE(),
'electronics_data'[.DaysIn2023] > 160 && 'electronics_data'[Quantity] < 2, 0.8, -- High risk
'electronics_data'[.DaysIn2023] <= 40 && 'electronics_data'[Quantity] >= 2, 0.1, -- Low risk
'electronics_data'[.DaysIn2023] > 40 && 'electronics_data'[Quantity] < 2, 0.5, -- Medium risk
0.5 -- Default medium risk for all other cases
)
5.ChurnStatusByDuration1 =
VAR ChurnDuration = SELECTEDVALUE('ChurnDuration'[Churn Duration], "90 days") -- Default to "90 days"
VAR MaxOrderDate = MAX('electronics_data'[Order Date]) -- Get the latest order date
RETURN
SWITCH(
TRUE(),
ChurnDuration = "30 days",
IF(
YEAR(MaxOrderDate) = 2023 && DATEDIFF(MaxOrderDate, DATE(2023, 12, 31), DAY) > 30,
"Churned",
"Active"
),
ChurnDuration = "60 days",
IF(
YEAR(MaxOrderDate) = 2023 && DATEDIFF(MaxOrderDate, DATE(2023, 12, 31), DAY) > 60,
"Churned",
"Active"
),
ChurnDuration = "90 days",
IF(
YEAR(MaxOrderDate) = 2023 && DATEDIFF(MaxOrderDate, DATE(2023, 12, 31), DAY) > 90,
"Churned",
"Active"
),
ChurnDuration = "120 days",
IF(
YEAR(MaxOrderDate) = 2023 && DATEDIFF(MaxOrderDate, DATE(2023, 12, 31), DAY) > 120,
"Churned",
"Active"
),
ChurnDuration = "150 days",
IF(
YEAR(MaxOrderDate) = 2023 && DATEDIFF(MaxOrderDate, DATE(2023, 12, 31), DAY) > 150,
"Churned",
"Active"
),
"Active"
)
6.Customer Segment =
IF(
'electronics_data'[.ChurnProbability2023] > 0.6,
"High Risk",
IF(
'electronics_data'[.ChurnProbability2023] > 0.3,
"Medium Risk",
"Low Risk"
)
)
7.DaysIn2023 =
IF(
YEAR('electronics_data'[Order Date]) = 2023,
DATEDIFF(DATE(2023, 1, 1), 'electronics_data'[Order Date], DAY),
BLANK()
)
8.GPM% = DIVIDE(SUM('electronics_data'[Profit]), SUM('electronics_data'[Sales]), 0)
9.Month Name =
FORMAT('electronics_data'[Order Date], "MMMM")
10.Total Active Customers 2023 =
CALCULATE(
COUNTROWS('electronics_data'),
FILTER(
'electronics_data',
'electronics_data'[.ChurnStatusByDuration1] = "Active" &&
YEAR('electronics_data'[Order Date]) = 2023
)
)
11.Total Sales =
CALCULATE(
SUM('electronics_data'[Sales]),
YEAR('electronics_data'[Order Date]) = 2023
)
